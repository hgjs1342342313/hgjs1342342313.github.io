Select distinct decode(tb.strm,'4231','2023-24 Sem 1','4232','2023-24 Sem 2','4238','2023-24 Sum Sem')as TERM
, tb.acad_career as "ACAD_CAREER"
,mp.SUBJECT || mp.CATALOG_NBR AS "COURSE CODE"
,mp.CLASS_SECTION AS "CLASS SECTION"
,tb.CLASS_NBR As "CLASS NUMBER"
--,ca.CRSE_ATTR_VALUE As "F2F lecture in alternate weeks"
,to_char(mp.START_DT,'yyyy-mm-dd')AS "START DATE"
,to_char(mp.END_DT,'yyyy-mm-dd') AS "END DATE"
,decode(MON,'Y','MON','') AS "MON"
,decode(TUES,'Y','TUE','') AS "TUE"
,decode(WED,'Y','WED','') AS "WED"
,decode(THURS,'Y','THU','') AS "THU"
,decode(FRI,'Y','FRI','') AS "FRI"
,decode(SAT,'Y','SAT','')AS "SAT"
,decode(SUN,'Y','SUN','')AS "SUN"
,ff.room AS "VENUE"
,decode(to_char(mp.MEETING_TIME_START,'hh24:mi'),'08:00',' ',to_char(mp.MEETING_TIME_START,'hh24:mi'))AS "START TIME"
,decode(to_char(mp.MEETING_TIME_END,'hh24:mi'),'08:25',' ',to_char(mp.MEETING_TIME_END,'hh24:mi')) AS "END TIME"
,cat.course_title_long AS "COURSE TITLE"
, org.descr AS "OFFER DEPT"
, a.name1 as "INSTRUCTOR"
from sysadm.ps_z_spcs_class_mp mp
left join sysadm.ps_facility_tbl ff on ff.facility_id = mp.facility_id,
sysadm.ps_class_tbl tb
left join z_tb_venue_tbl a on a.crse_id= tb.crse_id and a.strm= tb.strm and a.class_section= tb.class_section
left join (Select * From sysadm.PS_CLASS_ATTRIBUTE Where CRSE_ATTR = 'FTFA') ca on ca.CRSE_ID = tb.CRSE_ID And ca.STRM = tb.STRM And ca.CLASS_SECTION = tb.CLASS_SECTION,
sysadm.ps_crse_catalog cat,
sysadm.ps_crse_offer coff,
sysadm.ps_acad_org_tbl org
where Z_ACTV_TYPE_HOST_K='LEC'
and mp.class_section !=' '
and coff.crse_id = cat.crse_id
and coff.effdt = cat.effdt
and org.acad_org = coff.acad_org
and (Z_MODULE_HOST_KEY, mp.SUBJECT, mp.CATALOG_NBR) in
(select cc.crse_id, cc.subject, replace(cc.catalog_nbr,' ','0')
from sysadm.ps_crse_offer cc where cc.crse_id = mp.z_module_host_key
and cc.subject = mp.subject and replace(cc.catalog_nbr,' ','0') = mp.catalog_nbr
--and cc.acad_career in ('UG','UGME','UGDE')
)
and cat.crse_id = tb.crse_id
and cat.effdt >= '01-JUN-2023' and cat.effdt <'01-JUL-2024'
--and cat.effdt = '01-JUN-2021'
and cat.eff_status = 'A'
and mp.start_dt > '01-AUG-2023'
and tb.crse_id = mp.z_module_host_key
and tb.strm in('4231','4232','4238')
--and tb.strm = '4211'
and tb.class_section = mp.class_section
and tb.class_stat = 'A'
and org.effdt = (select max(org1.effdt) from sysadm.ps_acad_org_tbl org1 where org.acad_org= org1.acad_org)
order by 1, 2 desc,3,4,6,7,8,9,10,11,12,13,14
